Count plot by Gender

library(RColorBrewer)
pa <- palette(brewer.pal(n=1, name='Pastel2'))
par(mfrow = c(1,3))
    invisible(lapply(50:52, function(i)
      boxplot(master_df[,i]~master_df$gender,
              main= names(master_df[i]),
              xlab='',ylab='Count',
              col=c(pa)
             # bg='white'
      )
    )
    )
